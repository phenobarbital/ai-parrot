import folium
from geopy.distance import geodesic

# Coordinates for Boston and New York
boston_coords = (42.3601, -71.0589)
new_york_coords = (40.7128, -74.0060)

# Create base map centered between Boston and New York
m = folium.Map(
    location=[(boston_coords[0] + new_york_coords[0]) / 2, (boston_coords[1] + new_york_coords[1]) / 2],
    zoom_start=7,
    tiles='OpenStreetMap'
)

# Add start marker (Boston)
folium.Marker(
    location=boston_coords,
    popup='Boston (Start)',
    tooltip='Boston',
    icon=folium.Icon(color='green', icon='play')
).add_to(m)

# Add end marker (New York)
folium.Marker(
    location=new_york_coords,
    popup='New York (End)',
    tooltip='New York',
    icon=folium.Icon(color='red', icon='stop')
).add_to(m)

# Add polyline for the route
folium.PolyLine(
    locations=[boston_coords, new_york_coords],
    color='blue',
    weight=5,
    opacity=0.7
).add_to(m)

# Calculate total distance between Boston and New York
total_distance_miles = geodesic(boston_coords, new_york_coords).miles

# Add distance markers every 50 miles
segment_length_miles = 50
num_markers = int(total_distance_miles / segment_length_miles)

for i in range(1, num_markers + 1):
    fraction = (i * segment_length_miles) / total_distance_miles
    
    # Linear interpolation of latitude and longitude for intermediate points
    inter_lat = boston_coords[0] + (new_york_coords[0] - boston_coords[0]) * fraction
    inter_lon = boston_coords[1] + (new_york_coords[1] - boston_coords[1]) * fraction
    
    marker_location = (inter_lat, inter_lon)
    
    folium.Marker(
        location=marker_location,
        popup=f'{i * segment_length_miles} miles',
        tooltip=f'{i * segment_length_miles} miles from Boston',
        icon=folium.Icon(color='blue', icon='info-sign')
    ).add_to(m)
